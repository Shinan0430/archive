document.getElementById("zipInput").addEventListener("change", async (e) => { const file = e.target.files[0]; if (!file) return; showToast("読み込み中…"); try { const zip = await JSZip.loadAsync(file); // デバッグ（中身確認） console.log("ZIP中身:", Object.keys(zip.files)); let added = 0; let failed = 0; // ★ 全ファイル対象（フォルダ以外） const fileNames = Object.keys(zip.files) .filter(name => !zip.files[name].dir); for (const name of fileNames) { try { const fileObj = zip.files[name]; // バイナリ除外（軽くチェック） if (fileObj._data && fileObj._data.uncompressedSize === 0) continue; const content = await fileObj.async("string"); // ファイル名だけ取り出す const title = name.split("/").pop(); await sbUpsert(title, content || "（空ファイル）"); added++; } catch (e) { console.error("FILE ERROR:", name, e); failed++; } } if (added > 0) { showToast(`${added}件保存（${failed}件失敗）`, "success"); } else { showToast("保存できるファイルがなかった", "error"); } await loadList(); } catch (err) { console.error("ZIP ERROR:", err); showToast("ZIP読み込み失敗: " + err.message, "error"); } });
